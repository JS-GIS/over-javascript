<!-- 
    初始化目录结构案例
 -->

 ```js
const path = require('path');
const fs = require('fs');

let root = 'D:\\b';  
let fileContent =`
    <!DOCTYPE html>
        <head>
            <meta charset = "UTF-8">
            <title></title>
        </head>
        <body></body>
    </html>
`;

// 初始化数据
let initData = {
    projectName: 'mydemo',  // 根文件名
    data: [      // 里面的文件夹名称及文件名称
        {
            name: 'images',
            type: 'dir'
        },
        {
            name: 'css',
            type: 'dir'
        },
        {
            name: 'js',
            type: 'dir'
        },
        {
            name: 'index.html',
            type: 'file'
        }
    ]
}


// 创建项目根路径
fs.mkdir(path.join(root,initData.projectName),(err)=>{
    if(err) return;

    initData.data.forEach((item)=>{
        if(item.type == 'dir'){
            // 创建子目录
            fs.mkdirSync(path.join(root,initData.projectName, item.name));
        }else if(item.type == 'file'){
            // 创建文件并写入内容
            fs.writeFileSync(path.join(root,initData.projectName, item.name), fileContent);
        }
    })
})

 ```
