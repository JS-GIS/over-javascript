利用 ES5的 `Object.create()`函数可以实现对一个对象的继承：
```js
// 传入对象obj
function inherit(father) {

    if(father == null) {
        throw TypeError();
    }

    if (Object.create) {
        return Object.create(father);
    }

    var t = typeof father;
    if (t !== "object" ) {
        throw TypeError();
    }

    function f(){};
    f.prototype = father;
    return new f();

}

// 对象 father
var father = {
    surname:"李", 
    age:30,
};

// 对象son：继承了父对象的所有成员，也拥有自己的成员sex
var son = inherit(father);                                  // inherit是自定义的继承函数
son.sex = 1;
```


使用工厂方式配合上述函数也可以用来创建一个类，该方式不常用，只是一个笔记，过程如下：
- 定义一个原型对象
- 通过inherit()函数创建一个继承自它的对象，该对象即JS的一个类、
```js
function inherit(father) {

    if(father == null) {
        throw TypeError();
    }

    if (Object.create) {
        return Object.create(father);
    }

    var t = typeof father;
    if (t !== "object" ) {
        throw TypeError();
    }

    function f(){};
    f.prototype = father;
    return new f();

}

// 工厂函数
function person(name, age){

    var p = inherit(person.methods);
    p.name = name;
    p.age = age;

    return p;
}

person.methods = {
    sing: function(){
        console.log(this.name + " is sing...");
    }
}

var p = person("学友", 13);
p.sing();
console.log(p.age);
```

明显发现上述方式没有构造函数方式清晰直观，但是减少了 new 的过程。
